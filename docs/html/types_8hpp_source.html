157struct std::hash<automerge_cpp::ChangeHash> {

159 // First 8 bytes of the SHA-256 hash are already well-distributed
160 auto result = std::size_t{0};
161 const auto* p = reinterpret_cast<const unsigned char*>(ch.bytes.data());
162 for (std::size_t i = 0; i < sizeof(std::size_t); ++i) {
